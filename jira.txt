name: Repository Provisioning

on:
workflow_dispatch:
inputs:
repo_name:
required: true
description:
required: true
visibility:
required: true
default: private
admin_teams:
required: true
topics:
required: false
create_develop:
required: false
default: true

jobs:
provision:
runs-on: ubuntu-latest

```
steps:

  - uses: actions/checkout@v4

  - name: Create Repo From Template
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method POST \
        repos/YOUR_ORG/repo-template-standard/generate \
        -f owner=YOUR_ORG \
        -f name="${{ github.event.inputs.repo_name }}" \
        -f description="${{ github.event.inputs.description }}" \
        -f visibility=${{ github.event.inputs.visibility }} \
        -f include_all_branches=true \
        -f allow_auto_merge=true \
        -f allow_merge_commit=false \
        -f allow_rebase_merge=false \
        -f allow_squash_merge=true \
        -f delete_branch_on_merge=true

  - name: Wait Repo
    run: sleep 8

  - name: Assign Teams
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      IFS=',' read -ra TEAMS <<< "${{ github.event.inputs.admin_teams }}"
      for TEAM in "${TEAMS[@]}"; do
        gh api \
          --method PUT \
          orgs/YOUR_ORG/teams/$TEAM/repos/YOUR_ORG/${{ github.event.inputs.repo_name }} \
          -f permission=admin
      done

  - name: Apply Topics
    if: ${{ github.event.inputs.topics != '' }}
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method PUT \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/topics \
        -f names[]=${{ github.event.inputs.topics }}

  - name: Create Develop Branch
    if: ${{ github.event.inputs.create_develop == 'true' }}
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      SHA=$(gh api repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/git/ref/heads/main --jq '.object.sha')

      gh api \
        --method POST \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/git/refs \
        -f ref="refs/heads/develop" \
        -f sha="$SHA"

  - name: Create Environment
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method PUT \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/environments/dev

  - name: Set Repo Variables
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method POST \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/actions/variables \
        -f name=DEFAULT_REGION \
        -f value=us-east1

  - name: Enable Dependabot
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method PUT \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/automated-security-fixes

  - name: Set Actions Permission
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method PUT \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/actions/permissions \
        -f enabled=true \
        -f allowed_actions=selected

  - name: Import Ruleset
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method POST \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/rulesets \
        --input .github/rulesets/main-ruleset.json

  - name: Create Webhook
    env:
      GH_TOKEN: ${{ secrets.ORG_ADMIN_TOKEN }}
    run: |
      gh api \
        --method POST \
        repos/YOUR_ORG/${{ github.event.inputs.repo_name }}/hooks \
        -f config.url=https://your-ci-endpoint/webhook \
        -f config.content_type=json \
        -f events[]=push \
        -f events[]=pull_request

  - name: Summary
    run: |
      echo "Repo provisioned: ${{ github.event.inputs.repo_name }}" >> $GITHUB_STEP_SUMMARY
```

class PhotoService:
    @staticmethod
    def get_by_name(session, user: User, name: str) -> Photo:
        raw_sql = text(
            "SELECT * FROM photos WHERE name = :name AND user_id = :user_id"
        )

        return session.execute(
            raw_sql,
            {"name": name, "user_id": user.id}
        ).first()

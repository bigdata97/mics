- name: Run kubectl command
        env:
          KUBECTL_COMMAND: ${{ github.event.inputs.kubectl_command }}

          eval "$KUBECTL_COMMAND"

          echo "------------------------------------------------------------"
          echo "✅ kubectl command completed"
          echo "------------------------------------------------------------"

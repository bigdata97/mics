- name: Wait until repo is fully ready
  env:
    GH_TOKEN: ${{ steps.generate-token.outputs.token }}
  run: |
    for i in {1..20}; do
      echo "Checking repo readiness attempt $i"

      gh api repos/cvs-health-source-code/${{ github.event.inputs.repo_name }} \
        --jq '.default_branch' && break

      echo "Repo not ready yet..."
      sleep 5
    done

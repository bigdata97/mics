- name: Create Pull Request
  if: steps.commit-push.outputs.changes == 'true'
  env:
    GH_TOKEN: ${{ secrets.GHA_TOKEN }}
  run: |
    PR_URL=$(gh pr create \
      --base main \
      --head "$BRANCH_NAME_FOR_PR" \
      --title "Update image tag to ${TAG} (${ENV_NAME})" \
      --body "Automated image Tag update for ${ENV_NAME} environment."
    )

    echo "### 🔗 Pull Request Created" | tee -a "$GITHUB_STEP_SUMMARY"
    echo "[Open Pull Request]($PR_URL)" | tee -a "$GITHUB_STEP_SUMMARY"

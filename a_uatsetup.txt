update-manifest-pr:
  needs: docker-push
  if: github.ref_name == 'main' &&
      (github.event.inputs.deployment-environment == 'dev' ||
       github.event.inputs.deployment-environment == 'uat')
  runs-on: self-hosted

  permissions:
    contents: write
    pull-requests: write

  steps:

    # Always checkout latest main
    - name: Checkout main
      uses: actions/checkout@v4
      with:
        ref: main
        fetch-depth: 0

    - name: Sync with latest main
      run: |
        git fetch origin main
        git reset --hard origin/main

    - name: Set Variables
      run: |
        echo "IMAGE_TAG=${{ needs.docker-push.outputs.image_tag }}" >> $GITHUB_ENV
        echo "BRANCH_NAME=update-image-${{ github.run_id }}-$(date +%s)" >> $GITHUB_ENV

    # Select deployment file path
    - name: Set deployment path
      run: |
        if [ "${{ github.event.inputs.deployment-environment }}" = "dev" ]; then
          echo "DEPLOY_PATH=k8s-manifests/dev/deployment.yaml" >> $GITHUB_ENV
        elif [ "${{ github.event.inputs.deployment-environment }}" = "uat" ]; then
          echo "DEPLOY_PATH=k8s-manifests/uat/deployment.yaml" >> $GITHUB_ENV
        fi

    - name: Create new branch
      run: |
        git checkout -b $BRANCH_NAME

    - name: Install yq
      run: |
        sudo wget -q https://github.com/mikefarah/yq/releases/latest/download/yq_linux_amd64 -O /usr/local/bin/yq
        sudo chmod +x /usr/local/bin/yq

    - name: Update image tag
      run: |
        yq -i '
          .spec.template.spec.containers[0].image |=
          sub(":.*$"; ":" + env(IMAGE_TAG))
        ' $DEPLOY_PATH

    - name: Check if changes exist
      id: changes
      run: |
        if [[ -n "$(git status --porcelain)" ]]; then
          echo "changes=true" >> $GITHUB_OUTPUT
        else
          echo "changes=false" >> $GITHUB_OUTPUT
        fi

    - name: Commit changes
      if: steps.changes.outputs.changes == 'true'
      run: |
        git config user.name "github-actions"
        git config user.email "github-actions@github.com"
        git add $DEPLOY_PATH
        git commit -m "Update image tag to ${IMAGE_TAG} for ${{ github.event.inputs.deployment-environment }}"

    - name: Push branch
      if: steps.changes.outputs.changes == 'true'
      run: |
        git push origin $BRANCH_NAME

    - name: Create Pull Request
      if: steps.changes.outputs.changes == 'true'
      run: |
        gh pr create \
          --base main \
          --head $BRANCH_NAME \
          --title "Update image tag to ${IMAGE_TAG} (${{ github.event.inputs.deployment-environment }})" \
          --body "Automated image update for ${{ github.event.inputs.deployment-environment }} environment."
      env:
        GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}

    - name: Enable Auto Merge
      if: steps.changes.outputs.changes == 'true'
      run: |
        gh pr merge --auto --squash --delete-branch
      env:
        GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}

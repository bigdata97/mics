# Base image with Node 22
FROM node:22.21.1-alpine

# Set working directory
WORKDIR /app

# Build args
ARG VITE_APP_ENV
ARG JFROG_READER_TOKEN

# Set env
ENV VITE_APP_ENV=$VITE_APP_ENV

# Copy only package files first (better caching)
COPY package*.json ./

# Copy .npmrc (for registry + configs only, no token)
COPY .npmrc ./

# Install dependencies securely
RUN echo "//cvsh.jfrog.io/artifactory/api/npm/cvs-aetnahealth-npm/:_authToken=${JFROG_READER_TOKEN}" >> /root/.npmrc \
    && npm install \
    && rm -f /root/.npmrc

# Copy remaining source code
COPY . .

# Build project
RUN npm run build

# Expose port
EXPOSE 8080

# Start app
CMD ["npm", "run", "preview", "--", "--port", "8080", "--host", "0.0.0.0"]

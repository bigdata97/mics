- name: Create or Update Ruleset (Self Healing)
  env:
    GH_TOKEN: ${{ steps.app-token.outputs.token }}
    REPO: ${{ github.event.inputs.repo_name }}
  run: |

    RULESET_NAME="main-protection"

    echo "Checking if ruleset already exists..."

    RULESET_ID=$(gh api \
      repos/cvs-health-source-code/$REPO/rulesets \
      --jq ".rulesets[] | select(.name==\"$RULESET_NAME\") | .id")

    if [ -z "$RULESET_ID" ]; then

      echo "Ruleset not found → Creating new ruleset"

      gh api \
        --method POST \
        repos/cvs-health-source-code/$REPO/rulesets \
        --input .github/rulesets/main-ruleset.json

    else

      echo "Ruleset exists (ID: $RULESET_ID) → Updating ruleset"

      gh api \
        --method PATCH \
        repos/cvs-health-source-code/$REPO/rulesets/$RULESET_ID \
        --input .github/rulesets/main-ruleset.json

    fi

RUN --mount=type=secret,id=artifactory_token \
    TOKEN="$(cat /run/secrets/artifactory_token)" && \
    echo "registry=https://cvsh.jfrog.io/artifactory/api/npm/cvs-aetnahealth-npm/" > /root/.npmrc && \
    echo "always-auth=true" >> /root/.npmrc && \
    echo "//cvsh.jfrog.io/artifactory/api/npm/cvs-aetnahealth-npm/:_authToken=${TOKEN}" >> /root/.npmrc && \
    npm install && \
    rm -f /root/.npmrc

COPY . .

RUN npm run build

- name: Set Repo Visibility to Internal
  env:
    GH_TOKEN: ${{ steps.generate-token.outputs.token }}
  run: |
    for i in {1..20}; do
      echo "Attempt $i to set visibility"

      if gh api \
        --method PATCH \
        repos/cvs-health-source-code/${{ github.event.inputs.repo_name }} \
        -f visibility=internal; then

        echo "Visibility updated"
        exit 0
      fi

      echo "Still provisioning… waiting"
      sleep 5
    done

    echo "Failed to update visibility after retries"
    exit 1
